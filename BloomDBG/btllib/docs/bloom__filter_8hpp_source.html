439 inline bool
440 BloomFilter::contains(const uint64_t* hashes) const
441 {
442  for (unsigned i = 0; i < hash_num; ++i) {
443  const auto normalized = hashes[i] % array_bits;
444  const auto mask = BIT_MASKS[normalized % CHAR_BIT];

475 inline std::shared_ptr<cpptoml::table>
476 BloomFilter::parse_header(std::ifstream& file, const std::string& magic_string)
477 {
478  const std::string magic_with_brackets = std::string("[") + magic_string + "]";
479 
480  std::string line;
481  std::getline(file, line);
482  if (line != magic_with_brackets) {
483  log_error(
484  std::string("Magic string does not match (likely version mismatch)\n") +
485  "File magic string:\t" + line + "\n" + "Loader magic string:\t" +
486  magic_with_brackets);
487  std::exit(EXIT_FAILURE);
488  }
489 
490  /* Read bloom filter line by line until it sees "[HeaderEnd]"
491  which is used to mark the end of the header section and
492  assigns the header to a char array*/
493  std::string toml_buffer(line + '\n');
494  bool header_end_found = false;
495  while (bool(std::getline(file, line))) {
496  toml_buffer.append(line + '\n');
497  if (line == "[HeaderEnd]") {
498  header_end_found = true;
499  break;
500  }
501  }
502  if (!header_end_found) {
503  log_error("Pre-built bloom filter does not have the correct header end.");
504  std::exit(EXIT_FAILURE);
505  }
506 
507  // Send the char array to a stringstream for the cpptoml parser to parse
508  std::istringstream toml_stream(toml_buffer);
509  cpptoml::parser toml_parser(toml_stream);
510  const auto header_config = toml_parser.parse();
511 
512  // Obtain header values from toml parser and assign them to class members
513  return header_config->get_table(magic_string);
514 }
